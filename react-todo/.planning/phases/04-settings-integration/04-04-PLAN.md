       ```typescript
       // Filter tasks based on sidebar selections
       let filteredTasks = tasks;

       // Category filter (per D-01): if categories selected, show tasks matching ANY selected category
       if (selectedCategoryIds && selectedCategoryIds.length > 0) {
         filteredTasks = filteredTasks.filter((task) => {
           // Handle "Uncategorized" special value
           if (selectedCategoryIds.includes('__uncategorized__')) {
             if (!task.todoCategoryId) return true;
           }
           // Normal category match
           if (selectedCategoryIds.includes(task.todoCategoryId)) return true;
           return false;
         });
       }

       // Show completed filter (per D-20): hide completed by default
       if (showCompleted === false) {
         filteredTasks = filteredTasks.filter((task) => !task.isCompleted);
       }

       // Due date filter (per D-18): apply date range
       if (selectedDateRange) {
         const now = new Date();
         const today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
         const tomorrow = new Date(today);
         tomorrow.setDate(today.getDate() + 1);

         filteredTasks = filteredTasks.filter((task) => {
           if (!task.dueDt) {
             return selectedDateRange === 'no-date';
           }
           const dueDate = new Date(task.dueDt);
           switch (selectedDateRange) {
             case 'today':
               return dueDate >= today && dueDate < tomorrow;
             case 'this-week': {
               const dayOfWeek = now.getDay();
               const mondayOffset = dayOfWeek === 0 ? -6 : -(dayOfWeek - 1);
               const monday = new Date(today);
               monday.setDate(today.getDate() + mondayOffset);
               const sunday = new Date(monday);
               sunday.setDate(monday.getDate() + 6);
               sunday.setHours(23, 59, 59, 999);
               return dueDate >= monday && dueDate <= sunday;
             }
             case 'next-week': {
               const dayOfWeek = now.getDay();
               const mondayOffset = dayOfWeek === 0 ? -6 : -(dayOfWeek - 1);
               const monday = new Date(today);
               monday.setDate(today.getDate() + mondayOffset + 7);
               const sunday = new Date(monday);
               sunday.setDate(monday.getDate() + 6);
               sunday.setHours(23, 59, 59, 999);
               return dueDate >= monday && dueDate <= sunday;
             }
             case 'overdue':
               return dueDate < today;
             case 'no-date':
               return false; // Already handled above
             default:
               return true;
           }
         });
       }

       // Sort: completed tasks to bottom (per D-21)
       filteredTasks = [...filteredTasks].sort((a, b) => {
         if (a.isCompleted !== b.isCompleted) {
           return a.isCompleted ? 1 : -1;
         }
         return 0;
       });
       ```
