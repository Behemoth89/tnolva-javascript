      firstName: null,

      isAuthenticated: false,
      isLoading: true,

      setAuth: (payload: SetAuthPayload) => set({ ... }),

    }),
    {
      name: 'auth-storage',

        if (state) { state.isLoading = false; }

    }

Store structure:
```typescript
interface TaskStore {
  // State
  tasks: Task[];
  isLoading: boolean;
  error: string | null;
  
  // Actions
  fetchTasks: () => Promise<void>;
  createTask: (payload: CreateTaskPayload) => Promise<void>;
  updateTask: (payload: UpdateTaskPayload) => Promise<void>;
  deleteTask: (id: string) => Promise<void>;
  toggleTaskCompletion: (id: string) => Promise<void>;
  clearError: () => void;
}
```

Implementation details:
- Import apiClient from src/lib/apiClient.ts
- Import types from src/types/task.ts
- fetchTasks: GET /Task, set tasks array, handle errors
- createTask: POST /Task with payload, prepend to tasks array
- updateTask: PUT /Task/{id} with payload, update in array
- deleteTask: DELETE /Task/{id}, remove from array
- toggleTaskCompletion: PUT /Task/{id} with { isCompleted: !current }, update in array
- All async actions set isLoading true before API call, false after
- Error messages stored in error field, cleared on successful operation
- Use try/catch with error.message for user-friendly messages
- NO persist middleware needed (tasks should refresh from API on load)

Key patterns from CONTEXT.md:
- D-10: Dedicated Zustand store (mirrors useAuthStore pattern)
- D-11: Store handles loading states, error states, and data caching
- D-12: No React Query - Zustand + direct API calls
